def Bubble_Sort(uData, ASC):

    if ASC == True:
       
        Length = len(uData) - 1
        Sort = False

        while not Sort:
            Sort = True
            for i in range(0, Length):
                if uData[i] > uData[i+1]:
                    Sort = False
                    uData[i], uData[i+1] = uData[i+1],  uData[i]

def Quick_Sort(uData, ASC):

    if ASC == True:
        Length = len(uData)
        if Length <= 0:
            return uData
        else:
            Pivot = uData.pop()

            Items_Higher = []
            Items_Lower = []

            for Item in uData:
                if Item > Pivot:
                    Items_Higher.append(Item)

                else:
                    Items_Lower.append(Item)

            return Quick_Sort(Items_Lower, ASC) + [Pivot] + Quick_Sort(Items_Higher, ASC)

########################################
